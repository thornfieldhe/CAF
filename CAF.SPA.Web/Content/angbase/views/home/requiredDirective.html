Required Directive.

Bu sayfadaki direktive ve filter required ile lazy yüklenmiştir. (manyFilter.js,dateDirective.js)
app.js dosyasındaki configuration ile "/home/requiredDirective" route u için required ile "js/directives/dateDirective.js","js/filters/manyFilter" dosyalarının yüklenmesini istedik. 

    when("/home/requiredDirective", {
        templateUrl: "views/home/requiredDirective.html",
        controller: "HomeCtrl",
        resolve: {
            load: function ($q, $rootScope) {
                var defer = $q.defer();
                require(["js/controllers/homeController", "js/directives/dateDirective.js", "js/filters/manyFilter"], function () {
                    defer.resolve();
                    $rootScope.$apply();
                });
                return defer.promise;
            }
        }
    }).
